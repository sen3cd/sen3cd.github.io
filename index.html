Click to stop countdown

$('.countdown_dashboard').data('countdown').stop();

Click to start countdown

$('.countdown_dashboard').data('countdown').start();

Click to update diffSeconds

$('.countdown_dashboard').data('countdown').update({
  diff: 3000
});
or
$('.countdown_dashboard').data('countdown').update({
  year: 11,
  month: 11,
  day: 11,
  hour: 11,
  min: 11,
  sec: 11
});
